/**

 * @return {boolean}
 * @function
 * @export
 */
export const isPrimitive = a => {
  switch (typeof a) {
    case 'boolean':
    case 'number':
    case 'string':
    case 'symbol':
    case 'undefined':
      return true
    default:
      return a === null
  }
}
